    "import numpy as np\n",
    "import re\n",
    "import joblib\n",
    "import os\n",
    "from flask import Flask, request, jsonify, render_template_string\n",
    "from Bio.Seq import Seq\n",
    "from sklearn.ensemble import RandomForestClassifier\n",
    "from xgboost import XGBRegressor\n",
    "import warnings\n",
    "warnings.filterwarnings('ignore')\n",
    "\n",
    "app = Flask(__name__)\n",
    "os.makedirs(\"models\", exist_ok=True)\n",
    "\n",
    "# ===================== 密码子优化 =====================\n",
    "ECOLI_OPT_CODONS = {\n",
    "    'F': 'TTC', 'L': 'CTG', 'I': 'ATC', 'M': 'ATG', 'V': 'GTG',\n",
    "    'S': 'TCT', 'P': 'CCG', 'T': 'ACC', 'A': 'GCG', 'Y': 'TAC',\n",
    "    'H': 'CAC', 'Q': 'CAG', 'N': 'AAC', 'K': 'AAA', 'D': 'GAT',\n",
    "    'E': 'GAA', 'C': 'TGC', 'W': 'TGG', 'R': 'CGT', 'G': 'GGC'\n",
    "}\n",
    "\n",
    "class CodonOptimizer:\n",
    "    @staticmethod\n",
    "    def optimize(dna_seq):\n",
    "        seq = re.sub(r'[^ATCG]', '', dna_seq.upper())\n",
    "        if len(seq) % 3 != 0:\n",
    "            raise ValueError(\"CDS长度必须为3的倍数\")\n",
    "        prot = str(Seq(seq).translate())\n",
    "        optimized = [ECOLI_OPT_CODONS[aa] for aa in prot if aa in ECOLI_OPT_CODONS]\n",
    "        return ''.join(optimized)\n",
    "\n",
    "# ===================== 特征提取 =====================\n",
    "class FeatureExtractor:\n",
    "    RARE_CODONS = {\"ATA\", \"AGG\", \"AGA\", \"CGA\", \"CGG\", \"CTA\"}\n",
    "    @staticmethod\n",
    "    def extract(dna_seq):\n",
    "        seq = re.sub(r'[^ATCG]', '', dna_seq.upper())\n",
    "        if len(seq) % 3 != 0:\n",
    "            raise ValueError(\"CDS长度必须为3的倍数\")\n",
    "        prot = str(Seq(seq).translate())\n",
    "        L = len(seq)\n",
    "        gc = (seq.count('G')+seq.count('C'))/L\n",
    "        codons = [seq[i:i+3] for i in range(0,L,3)]\n",
    "        rare = sum(1 for c in codons if c in FeatureExtractor.RARE_CODONS)/len(codons)\n",
    "        hydro = sum(1 for aa in prot if aa in 'AVILMFYW')/len(prot)\n",
    "        return np.array([L, gc, rare, hydro])\n",
    "\n",
    "# ===================== 自动训练模型 =====================\n",
    "def get_trained_models():\n",
    "    try:\n",
    "        exp_model = joblib.load(\"models/exp.pkl\")\n",
    "        sol_model = joblib.load(\"models/sol.pkl\")\n",
    "    except:\n",
    "        print(\"训练模型中...\")\n",
    "        aa = list(ECOLI_OPT_CODONS.keys())\n",
    "        seqs, exps, sols = [], [], []\n",
    "        np.random.seed(42)\n",
    "        for _ in range(500):\n",
    "            p = ''.join(np.random.choice(aa, np.random.randint(100,300)))\n",
    "            dna = ''.join([ECOLI_OPT_CODONS[x] for x in p])\n",
    "            seqs.append(dna)\n",
    "            exps.append(np.clip(np.random.normal(0.6,0.2),0.1,1))\n",
    "            sols.append(1 if np.random.random()>0.3 else 0)\n",
    "        X = [FeatureExtractor.extract(s) for s in seqs]\n",
    "        exp_model = XGBRegressor(n_estimators=150)\n",
    "        sol_model = RandomForestClassifier(n_estimators=150)\n",
    "        exp_model.fit(X, exps)\n",
    "        sol_model.fit(X, sols)\n",
    "        joblib.dump(exp_model, \"models/exp.pkl\")\n",
    "        joblib.dump(sol_model, \"models/sol.pkl\")\n",
    "    return exp_model, sol_model\n",
    "\n",
    "exp_model, sol_model = get_trained_models()\n",
